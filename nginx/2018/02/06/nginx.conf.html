使用本地图片映射

root@42a151189ddf:/# cat /etc/nginx/sites-enabled/local.conf
server {
    listen 10011;
    server_name 192.168.161.82;
    access_log  /app/logs/nginx/access.log;
    error_log  /app/logs/nginx/error.log;
    tcp_nodelay on;
    client_max_body_size 50M;

    location /img/ {
        root /app/;
        autoindex on;
    }
}
